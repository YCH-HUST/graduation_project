# 4.4.1 概念结构设计

## (1) 各实体 ER 图（含属性）

### 用户实体（User）

```mermaid
erDiagram
    USER {
        int id PK "用户ID（主键）"
        string username "用户名（唯一）"
        string password "密码（哈希）"
        string role "角色: patient/doctor/admin"
        string full_name "真实姓名"
        string email "邮箱"
        string gender "性别: male/female"
        int age "年龄"
        string hospital "所属医院（医生）"
        string department "科室（医生）"
        string job_title "职称（医生）"
        int years_of_experience "从业年限（医生）"
        datetime date_joined "注册时间"
        boolean is_active "是否激活"
    }
```

### 病例实体（Case）

```mermaid
erDiagram
    CASE {
        uuid id PK "病例ID（UUID主键）"
        int patient_id FK "患者用户ID"
        int assigned_doctor_id FK "主诊医生ID"
        string status "状态: draft/running/pending_review/approved/rejected/failed"
        string chief_complaint_text "主诉文本"
        json questionnaire_json "问诊问卷JSON"
        datetime created_at "创建时间"
        datetime updated_at "最近更新时间"
    }
```

### 病例资源实体（CaseAsset）

```mermaid
erDiagram
    CASE_ASSET {
        int id PK "资源ID"
        uuid case_id FK "病例ID"
        string type "类型: raw_image/mask/heatmap/annotated"
        string file "文件路径"
        datetime created_at "创建时间"
    }
```

### 审核记录实体（Review）

```mermaid
erDiagram
    REVIEW {
        int id PK "审核ID"
        uuid case_id FK "病例ID"
        int doctor_id FK "审核医生ID"
        string decision "决定: approved/rejected/revise"
        json edited_syndrome_json "修订后证候JSON"
        json edited_prescription_json "修订后处方JSON"
        string note "审核备注"
        datetime created_at "审核时间"
    }
```

### AI流水线运行记录实体（PipelineRun）

```mermaid
erDiagram
    PIPELINE_RUN {
        int id PK "运行ID"
        uuid case_id FK "病例ID"
        string status "状态: queued/running/success/failed"
        int progress "进度 0-100"
        json yolo_result_json "YOLO推理结果"
        json nlp_result_json "NLP解析结果"
        json inference_result_json "辨证推理结果"
        string explanation_text "LLM解释文本"
        json timing_json "各阶段耗时"
        string error_message "错误信息"
        datetime created_at "创建时间"
    }
```

### 聊天消息实体（ChatMessage）

```mermaid
erDiagram
    CHAT_MESSAGE {
        int id PK "消息ID"
        int patient_id FK "所属患者ID"
        int sender_id FK "发送者用户ID"
        string content "消息内容"
        boolean is_read "是否已读"
        datetime created_at "发送时间"
    }
```

### 系统通知实体（Notification）

```mermaid
erDiagram
    NOTIFICATION {
        int id PK "通知ID"
        int recipient_id FK "接收者用户ID"
        string type "类型: new_case/case_approved/case_rejected/system"
        string title "通知标题"
        string content "通知内容"
        uuid related_case_id FK "关联病例ID（可空）"
        boolean is_read "是否已读"
        datetime created_at "创建时间"
    }
```

### 用药计划实体（MedicationPlan）

```mermaid
erDiagram
    MEDICATION_PLAN {
        int id PK "计划ID"
        uuid case_id FK "关联病例ID（唯一）"
        int patient_id FK "患者ID"
        boolean is_active "是否激活"
        date end_date "计划结束日期"
        datetime created_at "创建时间"
    }
```

### 用药打卡记录实体（MedicationLog）

```mermaid
erDiagram
    MEDICATION_LOG {
        int id PK "记录ID"
        int plan_id FK "用药计划ID"
        date date "打卡日期"
        string slot "时段: morning/afternoon/evening"
        boolean taken "是否已服药"
        string feedback "主观反馈文字"
        datetime created_at "创建时间"
    }
```

### 审计日志实体（AuditLog）

```mermaid
erDiagram
    AUDIT_LOG {
        int id PK "日志ID"
        int user_id FK "操作用户ID（可空）"
        string action "操作类型: login/case_create/pipeline_start/review_approve/review_reject"
        uuid case_id FK "关联病例ID（可空）"
        json details "详细信息JSON"
        datetime created_at "操作时间"
    }
```

## (2) 全局 ER 图（不画实体属性）

```mermaid
erDiagram
    USER ||--o{ CASE : "patient\n提交"
    USER ||--o{ CASE : "assigned_doctor\n负责"
    USER ||--o{ REVIEW : "doctor\n审核"
    USER ||--o{ CHAT_MESSAGE : "patient\n所属会话"
    USER ||--o{ CHAT_MESSAGE : "sender\n发送"
    USER ||--o{ NOTIFICATION : "接收"
    USER ||--o{ MEDICATION_PLAN : "拥有"
    USER ||--o{ AUDIT_LOG : "产生"

    CASE ||--o{ CASE_ASSET : "包含"
    CASE ||--o{ REVIEW : "被审核"
    CASE ||--o{ PIPELINE_RUN : "触发"
    CASE ||--|| MEDICATION_PLAN : "生成"
    CASE ||--o{ NOTIFICATION : "关联"
    CASE ||--o{ AUDIT_LOG : "关联"

    MEDICATION_PLAN ||--o{ MEDICATION_LOG : "记录"

    REVIEW {
        string decision "审核决定 (有属性)"
        json edited_syndrome_json "修订证候"
        json edited_prescription_json "修订处方"
        string note "审核备注"
    }
```
