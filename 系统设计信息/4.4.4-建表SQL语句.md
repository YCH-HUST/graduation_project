# 4.4.4 数据库建表 SQL 语句

> 数据库：MySQL 8.0，字符集：utf8mb4

---

## 1. 用户表（auth_user）

```sql
CREATE TABLE `auth_user` (
    `id`                   INT             NOT NULL AUTO_INCREMENT COMMENT '用户ID',
    `username`             VARCHAR(150)    NOT NULL COMMENT '用户名',
    `password`             VARCHAR(128)    NOT NULL COMMENT '密码（哈希）',
    `role`                 VARCHAR(10)     NOT NULL DEFAULT 'patient' COMMENT '角色: patient/doctor/admin',
    `full_name`            VARCHAR(100)    NOT NULL DEFAULT '' COMMENT '真实姓名',
    `email`                VARCHAR(254)    NOT NULL DEFAULT '' COMMENT '邮箱',
    `gender`               VARCHAR(10)     NOT NULL DEFAULT '' COMMENT '性别: male/female',
    `age`                  INT UNSIGNED    NULL COMMENT '年龄',
    `hospital`             VARCHAR(100)    NOT NULL DEFAULT '' COMMENT '所属医院',
    `department`           VARCHAR(20)     NOT NULL DEFAULT '' COMMENT '科室',
    `job_title`            VARCHAR(50)     NOT NULL DEFAULT '' COMMENT '职称',
    `years_of_experience`  INT UNSIGNED    NULL COMMENT '从业年限',
    `is_active`            TINYINT(1)      NOT NULL DEFAULT 1 COMMENT '是否激活',
    `is_staff`             TINYINT(1)      NOT NULL DEFAULT 0 COMMENT '是否员工',
    `is_superuser`         TINYINT(1)      NOT NULL DEFAULT 0 COMMENT '是否超管',
    `date_joined`          DATETIME        NOT NULL COMMENT '注册时间',
    `last_login`           DATETIME        NULL COMMENT '最后登录时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_username` (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';
```

---

## 2. 病例表（case）

```sql
CREATE TABLE `case` (
    `id`                    VARCHAR(36)  NOT NULL COMMENT '病例ID（UUID）',
    `patient_id`            INT          NOT NULL COMMENT '患者用户ID',
    `assigned_doctor_id`    INT          NULL COMMENT '主诊医生ID',
    `status`                VARCHAR(20)  NOT NULL DEFAULT 'draft'
                                         COMMENT '状态: draft/running/pending_review/approved/rejected/failed',
    `chief_complaint_text`  LONGTEXT     NOT NULL COMMENT '主诉文本',
    `questionnaire_json`    JSON         NULL COMMENT '问诊问卷',
    `created_at`            DATETIME     NOT NULL COMMENT '创建时间',
    `updated_at`            DATETIME     NOT NULL COMMENT '更新时间',
    PRIMARY KEY (`id`),
    KEY `idx_patient_id` (`patient_id`),
    KEY `idx_doctor_id` (`assigned_doctor_id`),
    CONSTRAINT `fk_case_patient`
        FOREIGN KEY (`patient_id`) REFERENCES `auth_user` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_case_doctor`
        FOREIGN KEY (`assigned_doctor_id`) REFERENCES `auth_user` (`id`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='病例表';
```

## 3. 病例资源表（case_asset）

```sql
CREATE TABLE `case_asset` (
    `id`          INT          NOT NULL AUTO_INCREMENT COMMENT '资源ID',
    `case_id`     VARCHAR(36)  NOT NULL COMMENT '病例ID',
    `type`        VARCHAR(20)  NOT NULL COMMENT '类型: raw_image/mask/heatmap/annotated',
    `file`        VARCHAR(200) NOT NULL COMMENT '文件存储路径',
    `created_at`  DATETIME     NOT NULL COMMENT '创建时间',
    PRIMARY KEY (`id`),
    KEY `idx_case_id` (`case_id`),
    CONSTRAINT `fk_asset_case`
        FOREIGN KEY (`case_id`) REFERENCES `case` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='病例资源表';
```

## 4. 审核记录表（review）

```sql
CREATE TABLE `review` (
    `id`                        INT          NOT NULL AUTO_INCREMENT COMMENT '审核ID',
    `case_id`                   VARCHAR(36)  NOT NULL COMMENT '病例ID',
    `doctor_id`                 INT          NOT NULL COMMENT '审核医生ID',
    `decision`                  VARCHAR(10)  NOT NULL COMMENT '审核决定: approved/rejected/revise',
    `edited_syndrome_json`      JSON         NULL COMMENT '修订后证候',
    `edited_prescription_json`  JSON         NULL COMMENT '修订后处方',
    `note`                      LONGTEXT     NOT NULL COMMENT '审核备注',
    `created_at`                DATETIME     NOT NULL COMMENT '审核时间',
    PRIMARY KEY (`id`),
    KEY `idx_case_id` (`case_id`),
    CONSTRAINT `fk_review_case`
        FOREIGN KEY (`case_id`) REFERENCES `case` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_review_doctor`
        FOREIGN KEY (`doctor_id`) REFERENCES `auth_user` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='审核记录表';
```

## 5. AI流水线运行记录表（pipeline_run）

```sql
CREATE TABLE `pipeline_run` (
    `id`                     INT          NOT NULL AUTO_INCREMENT COMMENT '运行ID',
    `case_id`                VARCHAR(36)  NOT NULL COMMENT '病例ID',
    `status`                 VARCHAR(10)  NOT NULL DEFAULT 'queued'
                                          COMMENT '状态: queued/running/success/failed',
    `progress`               INT          NOT NULL DEFAULT 0 COMMENT '进度 0~100',
    `yolo_result_json`       JSON         NULL COMMENT 'YOLO推理结果',
    `nlp_result_json`        JSON         NULL COMMENT 'NLP解析结果',
    `inference_result_json`  JSON         NULL COMMENT '辨证推理结果',
    `explanation_text`       LONGTEXT     NOT NULL COMMENT 'LLM解释文本',
    `timing_json`            JSON         NULL COMMENT '各阶段耗时',
    `error_message`          LONGTEXT     NULL COMMENT '错误信息',
    `created_at`             DATETIME     NOT NULL COMMENT '创建时间',
    PRIMARY KEY (`id`),
    KEY `idx_case_id` (`case_id`),
    CONSTRAINT `fk_pipeline_case`
        FOREIGN KEY (`case_id`) REFERENCES `case` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='AI流水线运行记录表';
```

## 6. 聊天消息表（chat_message）

```sql
CREATE TABLE `chat_message` (
    `id`          INT         NOT NULL AUTO_INCREMENT COMMENT '消息ID',
    `patient_id`  INT         NOT NULL COMMENT '所属患者ID',
    `sender_id`   INT         NOT NULL COMMENT '发送者用户ID',
    `content`     LONGTEXT    NOT NULL COMMENT '消息内容',
    `is_read`     TINYINT(1)  NOT NULL DEFAULT 0 COMMENT '是否已读',
    `created_at`  DATETIME    NOT NULL COMMENT '发送时间',
    PRIMARY KEY (`id`),
    KEY `idx_patient_id` (`patient_id`),
    KEY `idx_sender_id` (`sender_id`),
    CONSTRAINT `fk_chat_patient`
        FOREIGN KEY (`patient_id`) REFERENCES `auth_user` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_chat_sender`
        FOREIGN KEY (`sender_id`) REFERENCES `auth_user` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='聊天消息表';
```

## 7. 系统通知表（notification）

```sql
CREATE TABLE `notification` (
    `id`               INT          NOT NULL AUTO_INCREMENT COMMENT '通知ID',
    `recipient_id`     INT          NOT NULL COMMENT '接收者用户ID',
    `type`             VARCHAR(20)  NOT NULL COMMENT '类型: new_case/case_approved/case_rejected/system',
    `title`            VARCHAR(100) NOT NULL COMMENT '通知标题',
    `content`          LONGTEXT     NOT NULL COMMENT '通知内容',
    `related_case_id`  VARCHAR(36)  NULL COMMENT '关联病例ID',
    `is_read`          TINYINT(1)   NOT NULL DEFAULT 0 COMMENT '是否已读',
    `created_at`       DATETIME     NOT NULL COMMENT '创建时间',
    PRIMARY KEY (`id`),
    KEY `idx_recipient_id` (`recipient_id`),
    CONSTRAINT `fk_notif_recipient`
        FOREIGN KEY (`recipient_id`) REFERENCES `auth_user` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_notif_case`
        FOREIGN KEY (`related_case_id`) REFERENCES `case` (`id`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统通知表';
```

## 8. 用药计划表（medication_plan）

```sql
CREATE TABLE `medication_plan` (
    `id`          INT          NOT NULL AUTO_INCREMENT COMMENT '计划ID',
    `case_id`     VARCHAR(36)  NOT NULL COMMENT '关联病例ID',
    `patient_id`  INT          NOT NULL COMMENT '患者ID',
    `is_active`   TINYINT(1)   NOT NULL DEFAULT 1 COMMENT '是否激活',
    `end_date`    DATE         NULL COMMENT '计划结束日期',
    `created_at`  DATETIME     NOT NULL COMMENT '创建时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_case_id` (`case_id`),
    KEY `idx_patient_id` (`patient_id`),
    CONSTRAINT `fk_plan_case`
        FOREIGN KEY (`case_id`) REFERENCES `case` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_plan_patient`
        FOREIGN KEY (`patient_id`) REFERENCES `auth_user` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用药计划表';
```

## 9. 用药打卡记录表（medication_log）

```sql
CREATE TABLE `medication_log` (
    `id`          INT          NOT NULL AUTO_INCREMENT COMMENT '记录ID',
    `plan_id`     INT          NOT NULL COMMENT '用药计划ID',
    `date`        DATE         NOT NULL COMMENT '打卡日期',
    `slot`        VARCHAR(10)  NOT NULL COMMENT '时段: morning/afternoon/evening',
    `taken`       TINYINT(1)   NOT NULL DEFAULT 0 COMMENT '是否已服药',
    `feedback`    LONGTEXT     NOT NULL COMMENT '主观反馈',
    `created_at`  DATETIME     NOT NULL COMMENT '创建时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_plan_date_slot` (`plan_id`, `date`, `slot`),
    CONSTRAINT `fk_log_plan`
        FOREIGN KEY (`plan_id`) REFERENCES `medication_plan` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用药打卡记录表';
```

## 10. 审计日志表（audit_log）

```sql
CREATE TABLE `audit_log` (
    `id`          INT          NOT NULL AUTO_INCREMENT COMMENT '日志ID',
    `user_id`     INT          NULL COMMENT '操作用户ID',
    `action`      VARCHAR(20)  NOT NULL
                               COMMENT '操作类型: login/case_create/pipeline_start/review_approve/review_reject',
    `case_id`     VARCHAR(36)  NULL COMMENT '关联病例ID',
    `details`     JSON         NULL COMMENT '详细信息',
    `created_at`  DATETIME     NOT NULL COMMENT '操作时间',
    PRIMARY KEY (`id`),
    KEY `idx_user_id` (`user_id`),
    CONSTRAINT `fk_audit_user`
        FOREIGN KEY (`user_id`) REFERENCES `auth_user` (`id`) ON DELETE SET NULL,
    CONSTRAINT `fk_audit_case`
        FOREIGN KEY (`case_id`) REFERENCES `case` (`id`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='审计日志表';
```
