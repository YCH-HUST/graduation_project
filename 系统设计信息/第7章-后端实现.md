# 第7章 后端实现

本章重点阐述中医智能辅助诊疗系统后端的业务流引擎实现与核心功能落地。后端采用 Python 生态中成熟的 Django 框架，并结合 Django REST Framework 提供标准化的架构支撑，重点解决了大模型集成、异步任务调度及跨端并发访问的稳定性问题。本章将选取代表性的五个核心后端功能模块进行详细解析。

## 7.1 用户注册与登录认证模块

医疗类系统对数据互访隔离与操作追溯有严苛的要求。在标准的账号密码认证之外，后端系统采用基于无状态的 JWT（JSON Web Token）双令牌机制（Access Token 与 Refresh Token）维持长效安全的会话层。同时，我们在 Django 默认的用户权限系统之上扩展出了患者、医生、管理员三种角色标识。为了应对日益增长的医疗安全纠纷追溯需求，所有的登录请求一旦签发 Token，都会被内置的统一拦截器记入数据库的日志审计（AuditLog）内，实现核心操作的不可逆留痕。

以下截取了后端 `accounts` 模块中处理用户登录并派发鉴权 Token 的核心实现：

```python
# 截取自：backend/apps/accounts/views.py
from rest_framework import status
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework.permissions import AllowAny
from rest_framework_simplejwt.tokens import RefreshToken

from .serializers import LoginSerializer, UserSerializer
from apps.audit.utils import log_action

class LoginView(APIView):
    """
    用户登录接口 (POST /auth/login/)
    """
    permission_classes = [AllowAny] # 登录接口对外完全开放
    
    def post(self, request):
        serializer = LoginSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)
        
        # 提取校验通过的安全 User 实体
        user = serializer.validated_data['user']
        
        # 使用 SimpleJWT 库为当前用户对象签发非对称加密的 JWT Token 对
        refresh = RefreshToken.for_user(user)
        access_token = str(refresh.access_token)
        
        # 旁路记录安全审计日志，留存登录 IP 等网络指纹
        log_action(user=user, action='login', details={'ip': self.get_client_ip(request)})
        
        return Response({
            'token': access_token,
            'refresh': str(refresh),
            'role': user.role,
            'user': UserSerializer(user).data
        }, status=status.HTTP_200_OK)
    
    def get_client_ip(self, request):
        """防逆向获取经过 Nginx 等代理层转派的真实客户端 IP"""
        x_forwarded_for = request.META.get('HTTP_X_FORWARDED_FOR')
        if x_forwarded_for:
            return x_forwarded_for.split(',')[0]
        return request.META.get('REMOTE_ADDR', '')
```


## 7.2 核心 AI 智能辅助诊断流水线调度模块

本系统的核心难点在于跨越多个神经网络架构（YOLO 视觉模型、机器学习决策树与通用大语言模型）进行聚合推理，这是极度耗时的重负载作业。若采用传统的同步 HTTP 请求响应模型，必将导致 Web 网关框架超时。因此系统对于 AI 模型的推理任务进行了独立线程下放，并将其精细切分为四个必须顺次执行的 Pipeline（流水线）子任务阶段，期间不断打点更新 `progress` 字段供前端轮询动画。

- 第一阶段（CV Phase）：下发 YOLO 引擎提取舌体特征图像，耗时进度 10%->40%
- 第二阶段（NLP Phase）：下发大模型进行非结构化长文本（主诉病情）分析提取出结构化的中医症状集，耗时进度 40%->60%
- 第三阶段（Inference Phase）：使用中医机器学习特征分类器，统合视觉与文本节点，预测患者证候与推荐用药，耗时进度 60%->85%
- 第四阶段（Explanation Phase）：再次调用大模型将所有中医术语转化为白话文解释，最终组装返回，耗时进度 85%->100%

以下为负责调度这四段核心管道的后台处理代码：

## 7.3 业务边界网关与审核状态机引擎

在 AI 引擎处理完毕并交还报告后，患者核对无误将触发前端的独立接线口转交实际医生进行确认拦截。后端使用状态机约束以实现严格的网关控制。

以下为病例审核扭转路由控制阀的代码截取：

```python
# 截取自：backend/apps/cases/views.py
from rest_framework import viewsets, status
from rest_framework.response import Response
from rest_framework.decorators import action

from .models import Case
from apps.audit.utils import log_action
from apps.notifications.utils import notify_doctors_new_case

class CaseViewSet(viewsets.GenericViewSet):
    """病例核心业务调度视图集"""

    @action(detail=True, methods=['post'], url_path='submit-to-doctor')
    def submit_to_doctor(self, request, pk=None):
        """
        AI 管线解析完毕后，患者将报告转交真实医生审核
        状态机流转：running (AI已完工) -> pending_review (医生待审)
        """
        try:
            case = Case.objects.get(pk=pk)
        except Case.DoesNotExist:
            return Response({'detail': '病例实体丢失'}, status=status.HTTP_404_NOT_FOUND)

        # 严格的业务约束边界：防御越权和状态机错乱
        if request.user.role == 'patient' and case.patient != request.user:
            return Response({'detail': '越权越界访问'}, status=status.HTTP_403_FORBIDDEN)
        if case.status != 'running':
            return Response({'detail': '不满足提交流水线的前置状态节点'}, status=status.HTTP_400_BAD_REQUEST)

        # 挂载状态扭转操作并唤醒通知模块
        case.status = 'pending_review'
        case.save()
        notify_doctors_new_case(case)
        log_action(user=request.user, action='case_submit', case=case)

        return Response({'ok': True, 'status': 'pending_review'})
```

## 7.4 实时医患沟通与消息推送模块

在医患即时沟通环节，若采用高频定时长轮询势必给 Django 集群施加大量空转查询压力；而部署完整的 WebSockets/ASGI 又太过笨重且受限于某些小程序的原生网络隔离池。为取得最佳平衡，后端最终抛弃了传统心跳模式，针对性采用 Server-Sent Events（SSE）长链接搭配基于内存缓存的单机事件总线（Event Bus）分发机制。

其原理利用了 Django 的 `StreamingHttpResponse`。该响应对象在生命周期内使用循环挂起（Generator Yield），只有当底库中检测到属于当前聊天室的全局 Timestamp 突发变动时，才组装新的增量 JSON 包推进网络流的客户端挂机口中。

以下为推送医患实时聊天消息的独立路由房间组件实现：

```python
# 截取自：backend/apps/chat/views.py
import json
import time
from rest_framework.views import APIView
from django.http import StreamingHttpResponse
from django.core.cache import cache
from .models import ChatMessage
from .serializers import ChatMessageSerializer

class ChatStreamView(APIView):
    """
    负责保持单一会话的 Server-Sent Events 长链接房间通道
    一旦监听到新的 ORM Hook，将立即释放挂机并下发增量包裹
    """
    def get(self, request, patient_id):
        def event_stream():
            # 划定当前频道的广播订阅 ID 并对齐世界观时间戳
            cache_key = f'chat_update_{patient_id}'
            yield "data: {\"type\": \"connected\"}\n\n"
            
            last_timestamp = cache.get(cache_key) or time.time()
            if cache.get(cache_key) is None:
                cache.set(cache_key, last_timestamp, timeout=None)
            
            last_message_id = ChatMessage.objects.filter(patient_id=patient_id).last().id or 0

            # 无限挂起轮询守护池
            while True:
                time.sleep(1) # 限流控制，避免 CPU 100% 飙升
                current_timestamp = cache.get(cache_key)
                
                # 若检测出世界线变动，触发捞取游标后面的增量载荷
                if current_timestamp is None or current_timestamp != last_timestamp:
                    new_messages = ChatMessage.objects.filter(
                        patient_id=patient_id, 
                        id__gt=last_message_id
                    ).order_by('created_at')
                    
                    if new_messages.exists():
                        data = json.dumps({'type': 'new_messages', 'messages': ChatMessageSerializer(new_messages, many=True).data})
                        yield f"data: {data}\n\n"  # 规范发送 SSE 报文
                        last_message_id = new_messages.last().id 
                        
                    last_timestamp = current_timestamp
                else:
                    # 保活心跳 ping，防止被 Nginx 掐平
                    yield ": ping\n\n"

        # 发送特殊的关闭 Buffer 的 Header 保证推流可以实时刷出
        response = StreamingHttpResponse(event_stream(), content_type='text/event-stream')
        response['Cache-Control'] = 'no-cache'
        response['X-Accel-Buffering'] = 'no'
        return response
```

## 7.5 管理端超参配置与健康巡检群

作为系统底座，监控集群对内网数据库及大模型外网 API 是否出现波动也是工程中不可或缺的一环。系统在 Django 自身中内嵌了一套极轻量的探针阵列（Probe Array）。这个微服务健康检索服务可以在一次 HTTP 生命周期内，并发侦测数据库 MySQL 连通性、服务器本地 YOLO 权重文件（`.pt`）完整性及内存载入验证，并通过预发 Request 向云端硅基流动平台的私密模型网关校验当前的全局 Secret 环境变量。

以下展示了后端聚合检查引擎的设计逻辑层：

```python
# 截取自：backend/apps/adminops/views.py
import os
import time
import requests
from datetime import datetime
from rest_framework.views import APIView
from rest_framework.response import Response

class HealthCheckView(APIView):
    """
    服务健康检查聚合测试接口 (GET /api/admin/health/)
    """
    def get(self, request):
        """汇总各子群服务的并集体量包，抛出网络木桶短板"""
        services = [
            self._check_django(),
            self._check_database(),
            self._check_yolo_model(),  # 检查服务器本地 CV 依赖权重是否丢失
            self._check_ml_models(),   # 检查证型/中药模型 pkl 是否存在
            self._check_llm_api(),     # 自测发包校验 LLM 云端联通性
        ]
        overall_status = self._calculate_overall_status(services)
        return Response({'services': services, 'overall_status': overall_status})
    
    def _check_llm_api(self) -> dict:
        """探针：检查外部 LLM API 外网节点连通度及私钥有效性"""
        start_time = time.time()
        try:
            # 动态抽离当前保存在数据库的环境变量池（实现无重启热插拔重配模型）
            from apps.adminops.models import AIConfig
            api_url = AIConfig.get('llm_api_url')
            api_key = AIConfig.get('llm_api_key')
            model   = AIConfig.get('llm_model_name')
        except Exception:
            api_url, api_key, model = '', '', ''

        try:
            # 向目标外部平台发起极低能耗下的对话连线侦测
            resp = requests.post(
                api_url,
                headers={'Authorization': f'Bearer {api_key}', 'Content-Type': 'application/json'},
                json={
                    'model': model,
                    'messages': [{'role': 'user', 'content': 'ping'}],
                    'max_tokens': 5, # 将计费Token损耗截断在5字以内
                },
                timeout=10 # 最高挂起 10s 预先爆破避免级联雪崩
            )
            latency_ms = int((time.time() - start_time) * 1000)
            
            if resp.status_code == 200:
                return {'name': 'LLM API', 'status': 'healthy', 'latency_ms': latency_ms, 'message': '接口稳定'}
            else:
                return {'name': 'LLM API', 'status': 'unhealthy', 'latency_ms': latency_ms, 'message': resp.json().get('message')}
        except Exception as e:
            return {'name': 'LLM API', 'status': 'unhealthy', 'message': str(e)[:80]}
```
